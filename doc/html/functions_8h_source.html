48// Input a number between 'min' and 'max'.
49// If the user input 'Enter', the default value 'dflt' will be set.
50{

58 num = dflt;
59 break;
60 }
61 cin.putback(ch);
62 }
63
64 cin >> num;
65 if (!cin || num < min || num > max) {
66 cout << " > Invalid input. Please try again. ";
67 if (!cin) {
68 cin.clear();
69 cin.ignore(10, '\n');
70 }
71 } else {
72 cin.ignore(10, '\n');
73 break;
74 }
75 }
76}

158template<typename T>
159int find(const vector<T> &v, const int &begin, const int &end, const T &target)
160// We assume that vec has been sorted in ascending order.
161// if found, returns -1; otherwise, returns the position of the smallest element larger than "target"
162{
163 if (end - begin == 0) return 0;
164 if (end - begin == 1) {
165 if (v[begin] == target)
166 return -1;
167 else if (v[begin] < target)
168 return begin + 1;
169 else return begin;
170 }
171
172 const int middle = (end + begin) / 2;
173 if (v[middle] == target)
174 return -1;
175 if (v[middle] < target)
176 return find(v, middle, end, target);
177 return find(v, begin, middle, target);
178}

180template<typename T>
181int find(const vector<T> &v, const T &target) {
182 return find(v, 0, v.size(), target);
183}

185template<typename T, typename v_it>
186void merge_sort(const v_it begin, const v_it end, bool (*_compare)(const T &, const T &))
187// "end" is not included in the vector.
188{
189 if (end - begin <= 1) return;
190
191 const int half = (end - begin) / 2;
192 v_it middle = begin + half;
193 merge_sort(begin, middle, *_compare);
194 merge_sort(middle, end, *_compare);
195
196 vector<T> temp(begin, end);
197 const v_it t_begin = temp.begin();
198 const v_it t_middle = temp.begin() + half;
199 const v_it t_end = temp.end();
200 v_it t_pos1 = t_begin, t_pos2 = t_middle, pos = begin;
201
202 while (t_pos1 < t_middle && t_pos2 < t_end) {
203 if ((*_compare)(*t_pos1, *t_pos2))
204 *(pos++) = *(t_pos1++);
205 else
206 *(pos++) = *(t_pos2++);
207 }
208 while (pos < end) {
209 if (t_pos1 < t_middle)
210 *(pos++) = *(t_pos1++);
211 else // if(t_pos2 < t_end)
212 *(pos++) = *(t_pos2++);
213 }
214}
